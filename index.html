Usage as a lib

package main

import (
    "fmt"
    "github.com/mh-cbon/ignore-file/ignored"
)


func main() {

  ignore := ignored.Ignored{}
  if err := ignore.Load(".gitignore"); err != nil {
    fmt.Println(err)
    return nil
  }

  if err := ignore.Append(".git"); err != nil {
    fmt.Println(err)
    return nil
  }

  computed := ignore.ComputeDirectory(".")
  for _, l := range computed {
    fmt.Println(l)
  }
}
